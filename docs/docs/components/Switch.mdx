import { Switch } from 'minerva-ui';

```js
import { Switch } from 'minerva-ui';
```

```jsx
() => {
  const [checked, setChecked] = React.useState(false);
  return (
    <Switch
      htmlFor="example1"
      checked={checked}
      onChange={() => setChecked(!checked)}
    ></Switch>
  );
};
```

## Switch with size variant

```jsx
() => {
  const [checked, setChecked] = React.useState(false);
  return (
    <>
      <Switch
        switchSize="lg"
        htmlFor="example2"
        checked={checked}
        onChange={() => setChecked(!checked)}
      ></Switch>
    </>
  );
};
```
