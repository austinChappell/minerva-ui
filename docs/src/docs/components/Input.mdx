import { Input } from 'minerva-ui';

Extends native `<input />` component

```js
import { Input } from 'minerva-ui';
```

```jsx live
() => {
  const [name, setName] = React.useState('');
  return (
    <Input
      value={name}
      onChange={e => setName(e.target.value)}
      placeholder="Basic Input"
    />
  );
};
```
