# Examples

### Disabled Input
```jsx
() => {
  const [name, setName] = React.useState('')
  return (
    <Input
      disabled
      value={name}
      onChange={e => setName(e.target.value)}
      placeholder="Basic Input"
    />
  )
}
```

### Password Input
```jsx
() => {
  const [name, setName] = React.useState('')
  return (
    <Input
      type="password"
      value={name}
      onChange={e => setName(e.target.value)}
      placeholder="Basic Input"
    />
  )
}
