# Input Field

Basic Input Field component

```js
import { InputField } from 'minerva-ui';
import { Input } from 'minerva-ui"
```

```jsx
() => {
  const [name, setName] = React.useState('')
  const [error, setError] = React.useState('')

  return (<InputField
            label="Label"
            errorMsg={error}
            errorStyles={{ color: '#ff0000', size: '12px' }}
            >
            <Input
                placeholder="Basic Input"
                value={name}
                onBlur={!name ? () => setError('This field cannot be empty') : null}
                onChange={e => {
                setError('');
                setName(e.target.value);
                }}
            ></Input>
        </InputField>)
}
```

```jsx
() => {
  const [name, setName] = React.useState('')

            htmlFor="basicInput"

            requiredMarkerColor="tomato"
```
