# TagsInput

```js
import { TagsInput } from 'minerva-ui';
```

```jsx
() => {
  const [tags, setTags] = React.useState([]);

  const addTag = e => {
    const newTag = e.target.value;
    if (e.key === 'Enter' && newTag) {
    setTags(previousTags => [...previousTags, newTag ])
    }
  };

  return <TagsInput tags={tags} onKeyDown={addTag} />;
};
```
