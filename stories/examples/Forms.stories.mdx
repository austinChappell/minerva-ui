# Forms

```jsx
() => {
  const [username, setUsername] = React.useState('');
  const [usernameError, setUsernameError] = React.useState('');

  const [password, setPassword] = React.useState('');
  const [passwordError, setPasswordError] = React.useState('');

  return (
    <PseudoBox
      as="form"
      bg="gray.200"
      py={10}
      px={10}
      m="0 auto"
      width="320px"
      height="305px"
      borderRadius="4px"
      boxShadow="lg"
    >
      <Flex
        flexDirection="column"
        alignItems="center"
        justifyContent="space-around"
        height="70%"
        mb="24px"
      >
        <InputField
          label="Username"
          errorMsg={usernameError}
          fontWeight="700"
          fontSize="14px"
          color="gray.700"
        >
          <Input
            placeholder="Username"
            value={username}
            onChange={e => {
              setUsernameError('');
              setUsername(e.target.value);
            }}
            onBlur={
              !username
                ? () => setUsernameError('Please enter a username.')
                : null
            }
            boxShadow="lg"
            mb="8px"
          ></Input>
        </InputField>
        <InputField
          label="Password"
          errorMsg={passwordError}
          fontWeight="700"
          fontSize="14px"
          color="gray.700"
        >
          <Input
            type="password"
            placeholder="Password"
            value={password}
            onChange={e => {
              setPasswordError('');
              setPassword(e.target.value);
            }}
            onBlur={
              !password
                ? () => setPasswordError('Please enter a password.')
                : null
            }
            boxShadow="lg"
            mb="8px"
          ></Input>
        </InputField>
      </Flex>
      <Flex alignItems="center" justifyContent="space-between">
        <Button
          background="#551A8B"
          color="#fff"
          _hover={{
            backgroundColor: '#fff',
            color: '#551A8B',
            borderColor: '#551A8B',
          }}
          p="8px 16px"
          lineHeight="1.7"
          fontWeight="700"
          fontSize="16px"
          onClick={e => e.preventDefault()}
        >
          Sign In
        </Button>
        <Link
          href="#"
          color="#551A8B"
          fontWeight="700"
        >
          Forgot Password?
        </Link>
      </Flex>
    </PseudoBox>
  );
};
```
